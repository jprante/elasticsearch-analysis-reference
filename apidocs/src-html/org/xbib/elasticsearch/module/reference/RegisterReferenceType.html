001package org.xbib.elasticsearch.module.reference;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.settings.Settings;
006import org.elasticsearch.index.AbstractIndexComponent;
007import org.elasticsearch.index.Index;
008import org.elasticsearch.index.mapper.MapperService;
009import org.elasticsearch.index.settings.IndexSettings;
010import org.xbib.elasticsearch.index.mapper.reference.ReferenceMapper;
011
012public class RegisterReferenceType extends AbstractIndexComponent {
013
014    @Inject
015    public RegisterReferenceType(Index index, @IndexSettings Settings indexSettings,
016                                 MapperService mapperService, Client client) {
017        super(index, indexSettings);
018        mapperService.documentMapperParser().putTypeParser("ref", new ReferenceMapper.TypeParser(client, indexSettings));
019    }
020}
